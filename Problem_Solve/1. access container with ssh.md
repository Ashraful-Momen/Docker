# Connect to your container as root (you're already doing this correctly)
docker exec -it -u root laravel_app bash

# Inside the container, install SSH server
apt update
apt install -y openssh-server

# Configure SSH
echo 'PermitRootLogin yes' >> /etc/ssh/sshd_config
echo 'PasswordAuthentication yes' >> /etc/ssh/sshd_config

# Set root password
passwd root
# Enter your desired password

# Start SSH service
service ssh start

# Check if SSH is running
service ssh status

# Get container IP
hostname -I
