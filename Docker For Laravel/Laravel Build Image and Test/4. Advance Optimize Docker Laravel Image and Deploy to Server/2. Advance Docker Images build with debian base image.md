## 2. Optimized Multi-Stage Dockerfile (Debian-based)

```dockerfile
# ================================
# STAGE 1: Composer Dependencies
# ================================
FROM composer:2 AS vendor
WORKDIR /app

# Copy composer files
COPY composer.json composer.lock ./

# Install dependencies
RUN composer install \
    --no-dev \
    --no-interaction \
    --prefer-dist \
    --ignore-platform-reqs \
    --optimize-autoloader \
    --apcu-autoloader \
    --no-scripts \
    --no-cache

# ================================
# STAGE 2: Frontend Assets Build
# ================================
FROM node:18-slim AS frontend
WORKDIR /app

# Install system dependencies for node-gyp
RUN apt-get update && apt-get install -y \
    python3 \
    make \
    g++ \
    && rm -rf /var/lib/apt/lists/*

# Copy package files
COPY package*.json ./

# Install dependencies
RUN npm ci --only=production --no-audit

# Copy source files
COPY . .

# Build frontend assets
RUN npm run build 2>/dev/null || echo "No build script found, skipping"

# ================================
# STAGE 3: Production Image
# ================================
FROM php:8.3-fpm-slim AS production

# Set working directory
WORKDIR /var/www/html

# Install system dependencies
RUN apt-get update && apt-get install -y \
    nginx \
    supervisor \
    libfreetype6-dev \
    libjpeg62-turbo-dev \
    libpng-dev \
    libzip-dev \
    libicu-dev \
    libonig-dev \
    libxml2-dev \
    libpq-dev \
    mysql-client \
    curl \
    ca-certificates \
    && docker-php-ext-configure gd --with-freetype --with-jpeg \
    && docker-php-ext-install -j$(nproc) \
        pdo_mysql \
        pdo_pgsql \
        mbstring \
        zip \
        gd \
        intl \
        opcache \
        bcmath \
        xml \
        soap \
    && apt-get clean \
    && rm -rf /var/lib/apt/lists/* /tmp/* /var/tmp/*

# Install Redis extension
RUN pecl install redis \
    && docker-php-ext-enable redis \
    && rm -rf /tmp/pear

# Install APCu extension for better performance
RUN pecl install apcu \
    && docker-php-ext-enable apcu \
    && rm -rf /tmp/pear

# Create application user
RUN groupadd -g 1000 www \
    && useradd -u 1000 -ms /bin/bash -g www www

# Copy vendor dependencies from composer stage
COPY --from=vendor --chown=www:www /app/vendor ./vendor

# Copy built assets from frontend stage
COPY --from=frontend --chown=www:www /app/public/build ./public/build

# Copy application files
COPY --chown=www:www . .

# Copy configuration files
COPY ./docker/nginx/nginx.conf /etc/nginx/nginx.conf
COPY ./docker/php/php.ini /usr/local/etc/php/conf.d/app.ini
COPY ./docker/supervisor/supervisord.conf /etc/supervisor/conf.d/supervisord.conf

# Create required directories and set permissions
RUN mkdir -p \
        storage/logs \
        storage/framework/cache \
        storage/framework/sessions \
        storage/framework/views \
        bootstrap/cache \
        /var/log/supervisor \
        /var/log/nginx \
        /var/run \
    && chown -R www:www \
        storage \
        bootstrap/cache \
        /var/log/supervisor \
        /var/log/nginx \
        /var/run \
    && chmod -R 775 \
        storage \
        bootstrap/cache

# Generate optimized autoload files
RUN composer dump-autoload --optimize --apcu --no-dev

# Laravel optimizations (run as www user)
USER www
RUN php artisan config:cache || echo "Config cache skipped" \
    && php artisan route:cache || echo "Route cache skipped" \
    && php artisan view:cache || echo "View cache skipped"

# Switch back to root for service management
USER root

# Health check
HEALTHCHECK --interval=30s --timeout=10s --start-period=60s --retries=3 \
    CMD curl -f http://localhost/health || exit 1

# Expose port
EXPOSE 80

# Start services with supervisor
CMD ["/usr/bin/supervisord", "-c", "/etc/supervisor/conf.d/supervisord.conf"]
```

## 3. Production Docker Compose

```yaml
# docker-compose.prod.yml
version: '3.8'

services:
  app:
    build:
      context: .
      dockerfile: Dockerfile
      target: production
    image: laravel_instasure:v1.0.1
    container_name: laravel_app
    restart: unless-stopped
    ports:
      - "80:80"
      - "443:443"
    environment:
      APP_ENV: production
      APP_DEBUG: false
      LOG_CHANNEL: stderr
      DB_HOST: mysql
      DB_DATABASE: ${DB_DATABASE}
      DB_USERNAME: ${DB_USERNAME}
      DB_PASSWORD: ${DB_PASSWORD}
      REDIS_HOST: redis
      CACHE_DRIVER: redis
      SESSION_DRIVER: redis
      QUEUE_CONNECTION: redis
    volumes:
      - storage_data:/var/www/html/storage
      - ./ssl:/etc/nginx/ssl
    depends_on:
      mysql:
        condition: service_healthy
      redis:
        condition: service_started
    networks:
      - laravel_network
    deploy:
      resources:
        limits:
          memory: 1G
        reservations:
          memory: 512M

  mysql:
    image: mysql:8.0
    container_name: laravel_mysql
    restart: unless-stopped
    environment:
      MYSQL_ROOT_PASSWORD: ${DB_ROOT_PASSWORD}
      MYSQL_DATABASE: ${DB_DATABASE}
      MYSQL_USER: ${DB_USERNAME}
      MYSQL_PASSWORD: ${DB_PASSWORD}
    volumes:
      - mysql_data:/var/lib/mysql
      - ./database/backups:/docker-entrypoint-initdb.d:ro
    ports:
      - "3306:3306"
    command: >
      --default-authentication-plugin=mysql_native_password
      --innodb-buffer-pool-size=512M
      --innodb-log-file-size=128M
      --max_connections=200
      --sql_mode=STRICT_TRANS_TABLES,NO_ZERO_DATE,NO_ZERO_IN_DATE,ERROR_FOR_DIVISION_BY_ZERO
    healthcheck:
      test: ["CMD", "mysqladmin", "ping", "-h", "localhost", "-u", "root", "-p$${MYSQL_ROOT_PASSWORD}"]
      interval: 10s
      timeout: 5s
      retries: 10
      start_period: 30s
    networks:
      - laravel_network

  redis:
    image: redis:7-alpine
    container_name: laravel_redis
    restart: unless-stopped
    command: >
      redis-server 
      --appendonly yes 
      --maxmemory 256mb 
      --maxmemory-policy allkeys-lru
    volumes:
      - redis_data:/data
    ports:
      - "6379:6379"
    healthcheck:
      test: ["CMD", "redis-cli", "ping"]
      interval: 10s
      timeout: 3s
      retries: 3
    networks:
      - laravel_network

  # Optional: Queue Worker (if you use queues)
  queue:
    image: laravel_instasure:v1.0.1
    container_name: laravel_queue
    restart: unless-stopped
    environment:
      APP_ENV: production
      APP_DEBUG: false
      DB_HOST: mysql
      DB_DATABASE: ${DB_DATABASE}
      DB_USERNAME: ${DB_USERNAME}
      DB_PASSWORD: ${DB_PASSWORD}
      REDIS_HOST: redis
    command: php artisan queue:work --sleep=3 --tries=3 --daemon
    volumes:
      - storage_data:/var/www/html/storage
    depends_on:
      - mysql
      - redis
      - app
    networks:
      - laravel_network

  # Optional: Scheduler (if you use scheduled tasks)
  scheduler:
    image: laravel_instasure:v1.0.1
    container_name: laravel_scheduler
    restart: unless-stopped
    environment:
      APP_ENV: production
      APP_DEBUG: false
      DB_HOST: mysql
      DB_DATABASE: ${DB_DATABASE}
      DB_USERNAME: ${DB_USERNAME}
      DB_PASSWORD: ${DB_PASSWORD}
      REDIS_HOST: redis
    command: >
      sh -c "while true; do
        php artisan schedule:run --verbose --no-interaction &
        sleep 60
      done"
    volumes:
      - storage_data:/var/www/html/storage
    depends_on:
      - mysql
      - redis
      - app
    networks:
      - laravel_network

volumes:
  mysql_data:
    driver: local
  redis_data:
    driver: local
  storage_data:
    driver: local

networks:
  laravel_network:
    driver: bridge
```

```nginx
    client_header_buffer_size 1k;
    large_client_header_buffers 4 4k;
    output_buffers 1 32k;
    postpone_output 1460;
    
    # Gzip compression
    gzip on;
    gzip_vary on;
    gzip_min_length 1024;
    gzip_comp_level 6;
    gzip_types
        text/plain
        text/css
        text/xml
        text/javascript
        application/json
        application/javascript
        application/xml+rss
        application/atom+xml
        image/svg+xml;

    add_header X-Frame-Options "SAMEORIGIN" always;
    add_header X-Content-Type-Options "nosniff" always;
    add_header X-XSS-Protection "1; mode=block" always;
    add_header Referrer-Policy "strict-origin-when-cross-origin" always;
    
    # Hide Nginx version
    server_tokens off;
    
    # Rate limiting
    limit_req_zone $binary_remote_addr zone=api:10m rate=10r/s;
    limit_req_zone $binary_remote_addr zone=login:10m rate=1r/s;
    
    server {
        listen 80;
        listen [::]:80;
        server_name _;
        root /var/www/html/public;
        index index.php index.html;
        
        # Security configurations
        location ~ /\. {
            deny all;
            return 404;
        }
        
        # Laravel application
        location / {
            try_files $uri $uri/ /index.php?$query_string;
        }
        
        # PHP-FPM configuration
        location ~ \.php$ {
            try_files $uri =404;
            fastcgi_split_path_info ^(.+\.php)(/.+)$;
            fastcgi_pass 127.0.0.1:9000;
            fastcgi_index index.php;
            fastcgi_param SCRIPT_FILENAME $realpath_root$fastcgi_script_name;
            include fastcgi_params;
            
            # Security and performance
            fastcgi_hide_header X-Powered-By;
            fastcgi_read_timeout 300s;
            fastcgi_buffers 16 16k;
            fastcgi_buffer_size 32k;
        }
        
        # Static files caching
        location ~* \.(jpg|jpeg|gif|png|css|js|ico|webp|tiff|ttf|svg)$ {
            expires 1y;
            add_header Cache-Control "public, no-transform, immutable";
            access_log off;
            log_not_found off;
        }
        
        # API rate limiting
        location ^~ /api/ {
            limit_req zone=api burst=20 nodelay;
            try_files $uri $uri/ /index.php?$query_string;
        }
        
        # Login rate limiting
        location ^~ /login {
            limit_req zone=login burst=5 nodelay;
            try_files $uri $uri/ /index.php?$query_string;
        }
        
        # Health check endpoint
        location = /health {
            access_log off;
            return 200 "healthy\n";
            add_header Content-Type text/plain;
        }
        
        # Deny access to sensitive files
        location ~* \.(env|git|svn|htaccess|htpasswd)$ {
            deny all;
            return 404;
        }
        
        # Block common exploit attempts
        location ~* (wp-admin|wp-login|xmlrpc) {
            deny all;
            return 404;
        }
```

```bash
#!/bin/bash

# Build script for Laravel Docker (Debian Edition)
set -euo pipefail

# Configuration
IMAGE_NAME="yourusername/laravel_instasure"
VERSION="v1.0.1"
DOCKERFILE="Dockerfile"

# Colors for output
RED='\033[0;31m'
GREEN='\033[0;32m'
YELLOW='\033[1;33m'
BLUE='\033[0;34m'
NC='\033[0m' # No Color

# Functions
log_info() {
    echo -e "${BLUE}[INFO]${NC} $1"
}

log_success() {
    echo -e "${GREEN}[SUCCESS]${NC} $1"
}

log_warning() {
    echo -e "${YELLOW}[WARNING]${NC} $1"
}

log_error() {
    echo -e "${RED}[ERROR]${NC} $1"
}

# Check prerequisites
check_prerequisites() {
    log_info "Checking prerequisites..."
    
    if ! command -v docker &> /dev/null; then
        log_error "Docker is not installed or not in PATH"
        exit 1
    fi
    
    if ! docker info &> /dev/null; then
        log_error "Docker daemon is not running"
        exit 1
    fi
    
    if [ ! -f "$DOCKERFILE" ]; then
        log_error "Dockerfile not found in current directory"
        exit 1
    fi
    
    log_success "Prerequisites check passed"
}

# Build image
build_image() {
    log_info "Building Laravel Docker image..."
    log_info "Image: $IMAGE_NAME:$VERSION"
    
    # Enable BuildKit for better performance
    export DOCKER_BUILDKIT=1
    
    # Build with multi-platform support (optional)
    docker build \
        --platform linux/amd64 \
        --pull \
        --no-cache \
        --tag "$IMAGE_NAME:$VERSION" \
        --tag "$IMAGE_NAME:latest" \
        --build-arg BUILD_DATE="$(date -u +'%Y-%m-%dT%H:%M:%SZ')" \
        --build-arg VCS_REF="$(git rev-parse --short HEAD)" \
        --build-arg VERSION="$VERSION" \
        --file "$DOCKERFILE" \
        .
    
    if [ $? -eq 0 ]; then
        log_success "Image built successfully"
    else
        log_error "Image build failed"
        exit 1
    fi
}

# Test image
test_image() {
    log_info "Testing the built image..."
    
    # Basic container test
    docker run --rm --name laravel-test -d -p 8080:80 "$IMAGE_NAME:$VERSION"
    
    # Wait for container to start
    sleep 10
    
    # Test health endpoint
    if curl -f http://localhost:8080/health &> /dev/null; then
        log_success "Health check passed"
    else
        log_warning "Health check failed, but image seems to work"
    fi
    
    # Stop test container
    docker stop laravel-test || true
    
    log_success "Image test completed"
}

# Show image info
show_image_info() {
    log_info "Image information:"
    docker images "$IMAGE_NAME:$VERSION" --format "table {{.Repository}}\t{{.Tag}}\t{{.Size}}\t{{.CreatedAt}}"
    
    # Show layers (optional)
    log_info "Image layers:"
    docker history "$IMAGE_NAME:$VERSION" --format "table {{.CreatedBy}}\t{{.Size}}"
}

# Push to registry
push_image() {
    if [ "${SKIP_PUSH:-false}" = "true" ]; then
        log_warning "Skipping push to registry"
        return
    fi
    
    read -p "Push to Docker Hub? (y/N): " -n 1 -r
    echo
    
    if [[ $REPLY =~ ^[Yy]$ ]]; then
        log_info "Pushing to Docker Hub..."
        
        docker push "$IMAGE_NAME:$VERSION"
        docker push "$IMAGE_NAME:latest"
        
        log_success "Images pushed to Docker Hub"
        log_info "Pull command: docker pull $IMAGE_NAME:$VERSION"
    else
        log_info "Skipping push to registry"
    fi
}

# Cleanup
cleanup() {
    log_info "Cleaning up..."
    
    # Remove dangling images
    docker image prune -f
    
    log_success "Cleanup completed"
}

# Main execution
main() {
    log_info "🚀 Starting Laravel Docker build process..."
    
    check_prerequisites
    build_image
    test_image
    show_image_info
    push_image
    cleanup
    
    log_success "🎉 Build process completed successfully!"
    log_info "Docker image: $IMAGE_NAME:$VERSION"
    log_info "Image size: $(docker images --format 'table {{.Size}}' $IMAGE_NAME:$VERSION | tail -n 1)"
}

# Handle script arguments
while [[ $# -gt 0 ]]; do
    case $1 in
        --no-push)
            export SKIP_PUSH=true
            shift
            ;;
        --version)
            VERSION="$2"
            shift 2
            ;;
        --help)
            echo "Usage: $0 [OPTIONS]"
            echo "Options:"
            echo "  --no-push     Skip pushing to registry"
            echo "  --version     Specify version tag"
            echo "  --help        Show this help message"
            exit 0
            ;;
        *)
            log_error "Unknown option: $1"
            exit 1
            ;;
    esac
done

# Run main function
main "$@"
```

### deploy.sh
```bash
#!/bin/bash

# Deployment script for Laravel production server (Debian Edition)
set -euo pipefail

# Configuration
IMAGE_NAME="yourusername/laravel_instasure"
VERSION="v1.0.1"
APP_DIR="/var/www/laravel-app"
COMPOSE_FILE="docker-compose.prod.yml"
BACKUP_DIR="$APP_DIR/backups"

# Colors for output
RED='\033[0;31m'
GREEN='\033[0;32m'
YELLOW='\033[1;33m'
BLUE='\033[0;34m'
NC='\033[0m' # No Color

# Functions
log_info() {
    echo -e "${BLUE}[INFO]${NC} $1"
}

log_success() {
    echo -e "${GREEN}[SUCCESS]${NC} $1"
}

log_warning() {
    echo -e "${YELLOW}[WARNING]${NC} $1"
}

log_error() {
    echo -e "${RED}[ERROR]${NC} $1"
}

# Check if running as root or with sudo
check_privileges() {
    if [[ $EUID -eq 0 ]]; then
        log_warning "Running as root. Consider using a non-root user with docker permissions."
    fi
}

# Check prerequisites
check_prerequisites() {
    log_info "Checking prerequisites..."
    
    if ! command -v docker &> /dev/null; then
        log_error "Docker is not installed"
        exit 1
    fi
    
    if ! command -v docker-compose &> /dev/null; then
        log_error "Docker Compose is not installed"
        exit 1
    fi
    
    if [ ! -d "$APP_DIR" ]; then
        log_error "Application directory $APP_DIR does not exist"
        exit 1
    fi
    
    if [ ! -f "$APP_DIR/$COMPOSE_FILE" ]; then
        log_error "Docker Compose file not found: $APP_DIR/$COMPOSE_FILE"
        exit 1
    fi
    
    if [ ! -f "$APP_DIR/.env" ]; then
        log_error ".env file not found. Please copy .env.production to .env and configure it."
        exit 1
    fi
    
    log_success "Prerequisites check passed"
}

# Create backup
create_backup() {
    log_info "Creating backup..."
    
    # Create backup directory
    mkdir -p "$BACKUP_DIR"
    
    # Database backup
    BACKUP_FILE="$BACKUP_DIR/database-$(date +%Y%m%d-%H%M%S).sql"
    
    if docker-compose -f "$APP_DIR/$COMPOSE_FILE" exec -T mysql mysqladmin ping -h localhost --silent; then
        docker-compose -f "$APP_DIR/$COMPOSE_FILE" exec -T mysql mysqldump \
            -u root \
            -p"$DB_ROOT_PASSWORD" \
            "$DB_DATABASE" > "$BACKUP_FILE"
        
        if [ -f "$BACKUP_FILE" ]; then
            log_success "Database backup created: $BACKUP_FILE"
        else
            log_warning "Database backup may have failed"
        fi
    else
        log_warning "MySQL is not running, skipping database backup"
    fi
    
    # Keep only last 5 backups
    ls -t "$BACKUP_DIR"/database-*.sql 2>/dev/null | tail -n +6 | xargs -r rm --
}

# Pull latest image
pull_image() {
    log_info "Pulling latest image: $IMAGE_NAME:$VERSION"
    
    docker pull "$IMAGE_NAME:$VERSION"
    
    if [ $? -eq 0 ]; then
        log_success "Image pulled successfully"
    else
        log_error "Failed to pull image"
        exit 1
    fi
}

# Deploy application
deploy_application() {
    log_info "Deploying application..."
    
    cd "$APP_DIR"
    
    # Load environment variables
    set -a
    source .env
    set +a
    
    # Update image tag in compose file if needed
    export IMAGE_TAG="$VERSION"
    
    # Graceful deployment
    log_info "Stopping old containers..."
    docker-compose -f "$COMPOSE_FILE" down --timeout 30
    
    log_info "Starting new containers..."
    docker-compose -f "$COMPOSE_FILE" up -d --remove-orphans
    
    log_success "Containers started"
}

# Wait for services
wait_for_services() {
    log_info "Waiting for services to be ready..."
    
    # Wait for MySQL
    log_info "Checking MySQL connection..."
    for i in {1..30}; do
        if docker-compose -f "$APP_DIR/$COMPOSE_FILE" exec -T mysql mysqladmin ping -h localhost --silent 2>/dev/null; then
            log_success "MySQL is ready"
            break
        fi
        if [ $i -eq 30 ]; then
            log_error "MySQL failed to start"
            exit 1
        fi
        sleep 2
    done
    
    # Wait for application
    log_info "Checking application health..."
    for i in {1..20}; do
        if curl -f http://localhost/health &> /dev/null; then
            log_success "Application is ready"
            break
        fi
        if [ $i -eq 20 ]; then
            log_error "Application failed to start"
            exit 1
        fi
        sleep 3
    done
}

# Run post-deployment tasks
post_deployment() {
    log_info "Running post-deployment tasks..."
    
    cd "$APP_DIR"
    
    # Run database migrations
    log_info "Running database migrations..."
    if docker-compose -f "$COMPOSE_FILE" exec -T app php artisan migrate --force; then
        log_success "Database migrations completed"
    else
        log_warning "Database migrations may have failed"
    fi
    
    # Clear and warm up caches
    log_info "Optimizing application..."
    docker-compose -f "$COMPOSE_FILE" exec -T app php artisan optimize:clear
    docker-compose -f "$COMPOSE_FILE" exec -T app php artisan optimize
    
    # Fix permissions
    docker-compose -f "$COMPOSE_FILE" exec -T app chown -R www:www storage bootstrap/cache
    
    log_success "Post-deployment tasks completed"
}

# Health check
health_check() {
    log_info "Performing health check..."
    
    # Check container status
    if docker-compose -f "$APP_DIR/$COMPOSE_FILE" ps | grep -q "Up"; then
        log_success "Containers are running"
    else
        log_error "Some containers are not running"
        docker-compose -f "$APP_DIR/$COMPOSE_FILE" ps
        exit 1
    fi
    
    # Check HTTP endpoint
    if curl -f http://localhost/health &> /dev/null; then
        log_success "HTTP health check passed"
    else
        log_error "HTTP health check failed"
        exit 1
    fi
    
    # Check database connection
    if docker-compose -f "$APP_DIR/$COMPOSE_FILE" exec -T app php artisan tinker --execute="DB::connection()->getPdo(); echo 'DB OK';" | grep -q "DB OK"; then
        log_success "Database connection OK"
    else
        log_warning "Database connection check failed"
    fi
}

# Cleanup old images
cleanup() {
    log_info "Cleaning up old images..."
    
    # Remove old unused images
    docker image prune -f --filter "until=48h"
    
    # Remove old containers
    docker container prune -f
    
    log_success "Cleanup completed"
}

# Rollback function
rollback() {
    log_warning "Rolling back to previous version..."
    
    # This would require implementing version tracking
    # For now, we'll just restart current containers
    cd "$APP_DIR"
    docker-compose -f "$COMPOSE_FILE" restart
    
    log_info "Rollback completed. Please check application status."
}

# Show status
show_status() {
    log_info "Application Status:"
    
    cd "$APP_DIR"
    
    echo "=== Container Status ==="
    docker-compose -f "$COMPOSE_FILE" ps
    
    echo -e "\n=== Resource Usage ==="
    docker stats --no-stream $(docker-compose -f "$COMPOSE_FILE" ps -q)
    
    echo -e "\n=== Recent Logs ==="
    docker-compose -f "$COMPOSE_FILE" logs --tail=20
}

# Main deployment function
main() {
    log_info "🚀 Starting Laravel deployment process..."
    
    check_privileges
    check_prerequisites
    create_backup
    pull_image
    deploy_application
    wait_for_services
    post_deployment
    health_check
    cleanup
    
    log_success "🎉 Deployment completed successfully!"
    log_info "Application URL: http://$(hostname -I | awk '{print $1}')"
    log_info "Health endpoint: http://$(hostname -I | awk '{print $1}')/health"
}

# Handle script arguments
case "${1:-deploy}" in
    deploy)
        main
        ;;
    rollback)
        rollback
        ;;
    status)
        show_status
        ;;
    health)
        health_check
        ;;
    backup)
        create_backup
        ;;
    cleanup)
        cleanup
        ;;
    help|--help|-h)
        echo "Usage: $0 [COMMAND]"
        echo "Commands:"
        echo "  deploy     Deploy the application (default)"
        echo "  rollback   Rollback to previous version"
        echo "  status     Show application status"
        echo "  health     Run health check"
        echo "  backup     Create database backup"
        echo "  cleanup    Clean up old images and containers"
        echo "  help       Show this help message"
        exit 0
        ;;
    *)
        log_error "Unknown command: $1"
        log_info "Use '$0 help' to see available commands"
        exit 1
        ;;
esac
```

## 8. Key Differences: Debian vs Alpine

### Image Sizes:
- **Alpine-based**: ~200-250MB
- **Debian-based**: ~350-400MB (still much smaller than your 4.6GB!)

### Package Management:
- **Alpine**: `apk add` (lighter, fewer packages)
- **Debian**: `apt-get install` (more packages available, better compatibility)

### Benefits of Debian Edition:
- **Better compatibility** with PHP extensions
- **More stable** package ecosystem  
- **Easier debugging** (familiar tools)
- **Better support** for complex applications
- **Wider software availability**
