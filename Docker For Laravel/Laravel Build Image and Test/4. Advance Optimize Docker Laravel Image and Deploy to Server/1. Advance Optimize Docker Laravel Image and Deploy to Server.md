# Perfect Laravel Docker Setup

## 1. Optimized Multi-Stage Dockerfile

```dockerfile
# ================================
# STAGE 1: Composer Dependencies
# ================================
FROM composer:2 AS vendor
WORKDIR /app
COPY composer.json composer.lock ./
RUN composer install \
    --no-dev \
    --no-interaction \
    --prefer-dist \
    --ignore-platform-reqs \
    --optimize-autoloader \
    --apcu-autoloader \
    --no-scripts

# ================================
# STAGE 2: Frontend Assets (if you have them)
# ================================
FROM node:18-alpine AS frontend
WORKDIR /app
# Copy package files
COPY package*.json ./
# Install dependencies
RUN npm ci --only=production
# Copy source files
COPY . .
# Build assets
RUN npm run build 2>/dev/null || echo "No build script found"

# ================================
# STAGE 3: Production Image
# ================================
FROM php:8.3-fpm-alpine AS production

# Set working directory
WORKDIR /var/www/html

# Install system dependencies
RUN apk add --no-cache \
    nginx \
    supervisor \
    freetype-dev \
    libjpeg-turbo-dev \
    libpng-dev \
    libzip-dev \
    icu-dev \
    oniguruma-dev \
    mysql-client \
    && docker-php-ext-configure gd --with-freetype --with-jpeg \
    && docker-php-ext-install -j$(nproc) \
        pdo_mysql \
        mbstring \
        zip \
        gd \
        intl \
        opcache \
        bcmath

# Install Redis extension
RUN apk add --no-cache --virtual .build-deps $PHPIZE_DEPS \
    && pecl install redis \
    && docker-php-ext-enable redis \
    && apk del .build-deps

# Create application user
RUN addgroup -g 1000 www \
    && adduser -u 1000 -G www -s /bin/sh -D www

# Copy vendor dependencies from composer stage
COPY --from=vendor --chown=www:www /app/vendor ./vendor

# Copy built assets from frontend stage
COPY --from=frontend --chown=www:www /app/public/build ./public/build

# Copy application files
COPY --chown=www:www . .

# Copy configuration files
COPY ./docker/nginx/nginx.conf /etc/nginx/nginx.conf
COPY ./docker/php/php.ini /usr/local/etc/php/conf.d/app.ini
COPY ./docker/supervisor/supervisord.conf /etc/supervisor/conf.d/supervisord.conf

# Set proper permissions
RUN mkdir -p storage/logs storage/framework/{cache,sessions,views} bootstrap/cache \
    && chown -R www:www storage bootstrap/cache \
    && chmod -R 775 storage bootstrap/cache

# Generate optimized autoload files
RUN composer dump-autoload --optimize --apcu --no-dev

# Laravel optimizations
RUN php artisan config:cache \
    && php artisan route:cache \
    && php artisan view:cache

# Switch to non-root user
USER www

# Health check
HEALTHCHECK --interval=30s --timeout=3s --start-period=5s --retries=3 \
    CMD curl -f http://localhost/health || exit 1

# Expose port
EXPOSE 80

# Start services with supervisor
CMD ["/usr/bin/supervisord", "-c", "/etc/supervisor/conf.d/supervisord.conf"]
```

## 2. Production Docker Compose

```yaml
# docker-compose.prod.yml
version: '3.8'

services:
  app:
    build:
      context: .
      dockerfile: Dockerfile
      target: production
    image: laravel_instasure:v1.0.1
    container_name: laravel_app
    restart: unless-stopped
    ports:
      - "80:80"
    environment:
      APP_ENV: production
      APP_DEBUG: false
      DB_HOST: mysql
      DB_DATABASE: ${DB_DATABASE}
      DB_USERNAME: ${DB_USERNAME}
      DB_PASSWORD: ${DB_PASSWORD}
      REDIS_HOST: redis
      CACHE_DRIVER: redis
      SESSION_DRIVER: redis
      QUEUE_CONNECTION: redis
    volumes:
      - storage_data:/var/www/html/storage
    depends_on:
      mysql:
        condition: service_healthy
      redis:
        condition: service_started
    networks:
      - laravel_network

  mysql:
    image: mysql:8.0
    container_name: laravel_mysql
    restart: unless-stopped
    environment:
      MYSQL_ROOT_PASSWORD: ${DB_ROOT_PASSWORD}
      MYSQL_DATABASE: ${DB_DATABASE}
      MYSQL_USER: ${DB_USERNAME}
      MYSQL_PASSWORD: ${DB_PASSWORD}
    volumes:
      - mysql_data:/var/lib/mysql
      - ./database/backups:/docker-entrypoint-initdb.d
    ports:
      - "3306:3306"
    healthcheck:
      test: ["CMD", "mysqladmin", "ping", "-h", "localhost"]
      interval: 10s
      timeout: 5s
      retries: 5
    networks:
      - laravel_network

  redis:
    image: redis:7-alpine
    container_name: laravel_redis
    restart: unless-stopped
    command: redis-server --appendonly yes
    volumes:
      - redis_data:/data
    ports:
      - "6379:6379"
    networks:
      - laravel_network

volumes:
  mysql_data:
  redis_data:
  storage_data:

networks:
  laravel_network:
    driver: bridge
```

## 3. Configuration Files

### docker/nginx/nginx.conf
```nginx
user www;
worker_processes auto;
pid /var/run/nginx.pid;

events {
    worker_connections 1024;
    use epoll;
    multi_accept on;
}

http {
    include /etc/nginx/mime.types;
    default_type application/octet-stream;
    
    # Logging
    access_log /var/log/nginx/access.log;
    error_log /var/log/nginx/error.log warn;
    
    # Performance
    sendfile on;
    tcp_nopush on;
    tcp_nodelay on;
    keepalive_timeout 65;
    client_max_body_size 100M;
    
    # Gzip compression
    gzip on;
    gzip_vary on;
    gzip_min_length 1024;
    gzip_types text/css application/javascript text/javascript application/json image/svg+xml;
    
    # Security headers
    add_header X-Frame-Options "SAMEORIGIN" always;
    add_header X-Content-Type-Options "nosniff" always;
    add_header X-XSS-Protection "1; mode=block" always;
    
    server {
        listen 80;
        server_name _;
        root /var/www/html/public;
        index index.php;
        
        # Laravel routing
        location / {
            try_files $uri $uri/ /index.php?$query_string;
        }
        
        # PHP-FPM
        location ~ \.php$ {
            fastcgi_pass 127.0.0.1:9000;
            fastcgi_index index.php;
            fastcgi_param SCRIPT_FILENAME $realpath_root$fastcgi_script_name;
            include fastcgi_params;
            
            # Security
            fastcgi_hide_header X-Powered-By;
        }
        
        # Static files caching
        location ~* \.(jpg|jpeg|gif|png|css|js|ico|xml)$ {
            expires 1y;
            add_header Cache-Control "public, immutable";
        }
        
        # Health check endpoint
        location /health {
            access_log off;
            return 200 "healthy\n";
            add_header Content-Type text/plain;
        }
        
        # Security
        location ~ /\.ht {
            deny all;
        }
        
        location ~ /\. {
            deny all;
        }
    }
}
```

## 5. GitLab CI/CD Pipeline

### .gitlab-ci.yml
```yaml
stages:
  - test
  - build
  - deploy

variables:
  DOCKER_REGISTRY: index.docker.io
  IMAGE_NAME: yourusername/laravel_instasure
  DOCKER_DRIVER: overlay2
  MYSQL_DATABASE: laravel_test
  MYSQL_ROOT_PASSWORD: test_password

services:
  - docker:24-dind
  - mysql:8.0

before_script:
  - docker login -u $DOCKER_USERNAME -p $DOCKER_PASSWORD $DOCKER_REGISTRY

# Run tests
test:
  stage: test
  image: php:8.3-alpine
  variables:
    DB_HOST: mysql
    DB_DATABASE: laravel_test
    DB_USERNAME: root
    DB_PASSWORD: test_password
  before_script:
    - apk add --no-cache git unzip libpng-dev libzip-dev
    - docker-php-ext-install pdo_mysql zip gd
    - curl -sS https://getcomposer.org/installer | php -- --install-dir=/usr/local/bin --filename=composer
  script:
    - composer install --no-interaction --prefer-dist
    - cp .env.testing .env
    - php artisan key:generate
    - php artisan migrate --force
    - php artisan test
  only:
    - main
    - develop

# Build and push Docker image
build:
  stage: build
  image: docker:24
  script:
    - |
      if [[ "$CI_COMMIT_BRANCH" == "main" ]]; then
        TAG="v1.0.1"
        LATEST_TAG="latest"
      else
        TAG="$CI_COMMIT_SHORT_SHA"
        LATEST_TAG="develop"
      fi
    - echo "Building image with tag: $TAG"
    - docker build --pull -t $IMAGE_NAME:$TAG .
    - docker tag $IMAGE_NAME:$TAG $IMAGE_NAME:$LATEST_TAG
    - docker push $IMAGE_NAME:$TAG
    - docker push $IMAGE_NAME:$LATEST_TAG
    - echo "Image pushed: $IMAGE_NAME:$TAG"
  only:
    - main
    - develop

# Deploy to production
deploy:
  stage: deploy
  image: alpine:latest
  before_script:
    - apk add --no-cache openssh-client
    - eval $(ssh-agent -s)
    - echo "$PRODUCTION_PRIVATE_KEY" | tr -d '\r' | ssh-add -
    - mkdir -p ~/.ssh && chmod 700 ~/.ssh
    - ssh-keyscan $PRODUCTION_HOST >> ~/.ssh/known_hosts
  script:
    - |
      ssh $PRODUCTION_USER@$PRODUCTION_HOST << 'EOF'
        cd /var/www/laravel-app
        
        # Pull latest image
        docker login -u $DOCKER_USERNAME -p $DOCKER_PASSWORD
        docker pull yourusername/laravel_instasure:v1.0.1
        
        # Stop existing containers
        docker-compose -f docker-compose.prod.yml down
        
        # Start new containers
        docker-compose -f docker-compose.prod.yml up -d
        
        # Run migrations
        docker exec laravel_app php artisan migrate --force
        
        # Clear caches
        docker exec laravel_app php artisan config:cache
        docker exec laravel_app php artisan route:cache
        docker exec laravel_app php artisan view:cache
        
        # Cleanup old images
        docker image prune -f
      EOF
  environment:
    name: production
    url: https://yourapp.com
  when: manual
  only:
    - main

# Cleanup old images
cleanup:
  stage: deploy
  image: docker:24
  script:
    - docker system prune -af
  when: manual
  allow_failure: true
```

## 6. Build Scripts

### build.sh
```bash
#!/bin/bash

# Build script for Laravel Docker
set -e

echo "🚀 Building Laravel Docker Image..."

# Build the image
docker build -t laravel_instasure:v1.0.1 .

# Tag for Docker Hub
docker tag laravel_instasure:v1.0.1 yourusername/laravel_instasure:v1.0.1
docker tag laravel_instasure:v1.0.1 yourusername/laravel_instasure:latest

# Push to Docker Hub
echo "📤 Pushing to Docker Hub..."
docker push yourusername/laravel_instasure:v1.0.1
docker push yourusername/laravel_instasure:latest

echo "✅ Build and push completed!"
echo "Image size:"
docker images laravel_instasure:v1.0.1
```

### deploy.sh
```bash
#!/bin/bash

# Deployment script for production server
set -e

echo "🚀 Deploying Laravel Application..."

# Pull latest image
docker pull yourusername/laravel_instasure:v1.0.1

# Create .env from template if not exists
if [ ! -f .env ]; then
    cp .env.production .env
    echo "⚠️  Please update .env file with your production values"
    exit 1
fi

# Stop existing containers
docker-compose -f docker-compose.prod.yml down

# Start new containers
docker-compose -f docker-compose.prod.yml up -d

# Wait for services to be ready
echo "⏳ Waiting for services to start..."
sleep 10

# Run database migrations
docker exec laravel_app php artisan migrate --force

# Optimize Laravel
docker exec laravel_app php artisan optimize

echo "✅ Deployment completed!"
echo "🌐 Application is running at: http://your-server-ip"
```
