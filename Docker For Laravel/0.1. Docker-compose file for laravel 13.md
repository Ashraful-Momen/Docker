services:
  postgres:
    image: postgis/postgis:16-3.4
    container_name: shtil-postgres
    environment:
      POSTGRES_DB: shtil
      POSTGRES_USER: shtil
      POSTGRES_PASSWORD: shtil123
    ports:
      - "5432:5432"
    volumes:
      - ./data_backup/postgres:/var/lib/postgresql/data
      - ./init-scripts:/docker-entrypoint-initdb.d
  redis:
    image: redis:7-alpine
    container_name: shtil-redis
    ports:
      - "6379:6379"
    volumes:
      - ./data_backup/redis:/data
    command: redis-server --appendonly yes
  pgadmin:
    image: dpage/pgadmin4:latest
    container_name: shtil-pgadmin
    environment:
      PGADMIN_DEFAULT_EMAIL: admin@shtil.com.bd
      PGADMIN_DEFAULT_PASSWORD: admin123
    ports:
      - "5050:80"
    volumes:
      - ./data_backup/pgadmin:/var/lib/pgadmin
    depends_on:
      - postgres
  laravel:
    build:
      context: ./backend
      dockerfile: Dockerfile
    container_name: shtil-laravel
    working_dir: /var/www/html
    volumes:
      - ./backend:/var/www/html
    ports:
      - "8000:8000"
    depends_on:
      - postgres
      - redis
    command: php artisan serve --host=0.0.0.0 --port=8000
