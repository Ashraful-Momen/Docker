# Docker Swarm Cheat Sheet 🚀

## Quick Setup
```bash
# Initialize swarm
docker swarm init --advertise-addr <IP>

# Join as worker
docker swarm join --token <TOKEN> <MANAGER-IP>:2377

# Get join tokens
docker swarm join-token worker
docker swarm join-token manager
```

## Nodes
```bash
# List nodes
docker node ls

# Inspect node
docker node inspect <NODE>

# Promote/demote
docker node promote <NODE>
docker node demote <NODE>

# Drain node (maintenance)
docker node update --availability drain <NODE>

# Activate node
docker node update --availability active <NODE>

# Label node
docker node update --label-add env=prod <NODE>
```

## Services
```bash
# Create service
docker service create --name web nginx

# With replicas
docker service create --name web --replicas 3 nginx

# With port
docker service create --name web --publish 8080:80 nginx

# List services
docker service ls

# Scale service
docker service scale web=5

# Update service
docker service update --image nginx:1.20 web

# View service details
docker service ps web

# Service logs
docker service logs web

# Remove service
docker service rm web
```

## Advanced Service Options
```bash
# Resource limits
docker service create \
  --name web \
  --limit-cpu 0.5 \
  --limit-memory 512M \
  --reserve-cpu 0.25 \
  --reserve-memory 256M \
  nginx

# Placement constraints
docker service create \
  --name web \
  --constraint 'node.role == worker' \
  --constraint 'node.labels.env == prod' \
  nginx

# Rolling updates
docker service create \
  --name web \
  --update-parallelism 2 \
  --update-delay 10s \
  --update-failure-action rollback \
  nginx

# Health check
docker service create \
  --name web \
  --health-cmd "curl -f http://localhost" \
  --health-interval 30s \
  nginx
```

## Stacks (Compose Files)
```bash
# Deploy stack
docker stack deploy -c docker-compose.yml myapp

# List stacks
docker stack ls

# Stack services
docker stack services myapp

# Stack tasks
docker stack ps myapp

# Remove stack
docker stack rm myapp
```

## Quick Compose Template
```yaml
version: '3.8'
services:
  web:
    image: nginx
    ports:
      - "80:80"
    deploy:
      replicas: 3
      restart_policy:
        condition: on-failure
      resources:
        limits:
          cpus: '0.5'
          memory: 512M
    networks:
      - frontend

  api:
    image: myapp:latest
    deploy:
      replicas: 2
    environment:
      - DB_HOST=db
    networks:
      - frontend
      - backend

  db:
    image: postgres:13
    environment:
      - POSTGRES_PASSWORD_FILE=/run/secrets/db_pass
    secrets:
      - db_pass
    volumes:
      - db_data:/var/lib/postgresql/data
    networks:
      - backend

networks:
  frontend:
    driver: overlay
  backend:
    driver: overlay

volumes:
  db_data:

secrets:
  db_pass:
    file: ./db_password.txt
```

## Networks
```bash
# Create overlay network
docker network create --driver overlay mynet

# Encrypted network
docker network create --driver overlay --opt encrypted mynet

# List networks
docker network ls

# Inspect network
docker network inspect mynet

# Connect service to network
docker service update --network-add mynet web
```

## Secrets & Configs
```bash
# Create secret
echo "password123" | docker secret create db_pass -
docker secret create ssl_cert ./cert.pem

# List secrets
docker secret ls

# Use in service
docker service create \
  --name db \
  --secret db_pass \
  --env POSTGRES_PASSWORD_FILE=/run/secrets/db_pass \
  postgres

# Create config
docker config create nginx_conf ./nginx.conf

# Use in service
docker service create \
  --name web \
  --config source=nginx_conf,target=/etc/nginx/nginx.conf \
  nginx
```

## Volumes
```bash
# Create volume
docker volume create myvolume

# Service with volume
docker service create \
  --name db \
  --mount source=myvolume,target=/var/lib/mysql \
  mysql

# NFS volume
docker volume create \
  --driver local \
  --opt type=nfs \
  --opt o=addr=192.168.1.100,rw \
  --opt device=:/path/to/share \
  nfs_volume
```

## Monitoring & Debugging
```bash
# System info
docker system info

# Service logs
docker service logs -f web

# Service inspect
docker service inspect web

# Node resources
docker node inspect <NODE> --format '{{.Description.Resources}}'

# System events
docker system events --filter type=service

# Stats
docker stats

# Clean up
docker system prune -a
```

## Common Troubleshooting
```bash
# Check swarm status
docker info | grep Swarm

# Node connectivity
ping <NODE-IP>
telnet <NODE-IP> 2377

# Service not starting
docker service ps web --no-trunc

# Check constraints
docker service inspect web --format '{{.Spec.TaskTemplate.Placement}}'

# Network issues
docker network ls
docker run --rm --net mynet alpine ping web
```

## Production Quick Setup
```bash
# 1. Firewall (Ubuntu)
sudo ufw allow 2377/tcp
sudo ufw allow 7946
sudo ufw allow 4789/udp

# 2. Initialize with labels
docker swarm init
docker node update --label-add env=prod <NODE>

# 3. Deploy monitoring
curl -o monitoring.yml https://raw.githubusercontent.com/stefanprodan/swarmprom/master/docker-compose.yml
docker stack deploy -c monitoring.yml monitoring

# 4. Basic security
export DOCKER_CONTENT_TRUST=1
```

## One-Liner Deploys
```bash
# Quick web server
docker service create --name web --publish 80:80 --replicas 3 nginx

# Database with persistence
docker service create --name db --mount source=db_vol,target=/var/lib/mysql -e MYSQL_ROOT_PASSWORD=secret mysql:8.0

# Redis cache
docker service create --name cache --publish 6379:6379 redis:alpine

# Quick monitoring
docker service create --name portainer --publish 9000:9000 --mount type=bind,src=/var/run/docker.sock,dst=/var/run/docker.sock portainer/portainer-ce
```

## Emergency Commands
```bash
# Force new cluster (lost quorum)
docker swarm init --force-new-cluster

# Force remove node
docker node rm --force <NODE>

# Emergency scale down
docker service scale web=0

# Quick rollback
docker service rollback web

# Leave swarm
docker swarm leave --force
```

## Quick Health Checks
```bash
# All services healthy?
docker service ls --format "table {{.Name}}\t{{.Replicas}}"

# Quick cluster status
docker node ls --format "table {{.Hostname}}\t{{.Status}}\t{{.Availability}}\t{{.ManagerStatus}}"

# Resource usage
docker stats --no-stream --format "table {{.Container}}\t{{.CPUPerc}}\t{{.MemUsage}}"
```

---

**Remember:** 
- Always use odd number of managers (3, 5, 7)
- Label your nodes for better service placement
- Use secrets for passwords, configs for files
- Monitor everything with Prometheus + Grafana
- Test updates in staging first!
